using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Ass5
{
    class calculater
    {
        public static int square(int x)
        {
            return x * x;
        }
        public static int cube(int x)
        {
            return x * x * x;
        }
        public static int round(double x)
        {
            int y = (int)(x);
            return y;
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine(calculater.square(10));
            Console.WriteLine(calculater.cube(100));
            Console.WriteLine(calculater.round(10.5));
        }
    }
}
