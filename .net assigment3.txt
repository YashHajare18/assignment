namespace Ass5
{
    class account
    {
        private string name;
        private int id;
        private int bal;
        
        public account(string name)
        {
            this.name = name;
            this.id = ++id;
            
        }
        public void deposit(int amt) 
        { 
            bal = bal + amt; 
        }
        public void withdraw(int amt)
        {
            if (bal > 0)
            { 
               bal = bal - amt;
            }
        }
        public void display()
        {
            Console.WriteLine($"NAME={name}  ID={id}  BALANCE={bal}");
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            account s = new account("yash");
            s.deposit(10000);
            s.withdraw(2000);
            s.display();
